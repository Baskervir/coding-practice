### 소스 코드

import java.util.*;

public class Main {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        for (int i = 0; i < 3; i++) {
            int zeroCount = 0;
            for (int j = 0; j < 4; j++) {
                if (sc.nextInt() == 0) {
                    zeroCount++;
                }
            }
            System.out.println(getResult(zeroCount));
        }
        sc.close();
    }

    private static String getResult(int zeroCount) {
        switch (zeroCount) {
            case 1: return "A";
            case 2: return "B";
            case 3: return "C";
            case 4: return "D";
            case 0: return "E";
            default: return "";
        }
    }
}
