import java.util.*;

public class Main {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        
        int n = sc.nextInt();
        String[] arr = new String[n];
        
        for (int i = 0; i < n; i++) {
            arr[i] = sc.next();
        }
        
        for (int i = 0; i < n; i++) {
            int totalScore = 0;
            int currentScore = 0;
            
            for (int j = 0; j < arr[i].length(); j++) {
                if (arr[i].charAt(j) == 'O') {
                    currentScore++;
                    totalScore += currentScore;
                } else {
                    currentScore = 0;
                }
            }
            
            System.out.println(totalScore);
        }
        sc.close();
    }
}
