import java.util.*;

public class Main {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        
        int[] len = new int[3];
        len[0] = sc.nextInt();
        len[1] = sc.nextInt();
        len[2] = sc.nextInt();
        
        Arrays.sort(len);
        
        int x = len[0];
        int y = len[1];
        int z = len[2];
        
        int max;
        
        if (x + y > z) {
            max = x + y + z;
        } else {
            max = x + y + (x + y - 1);
        }
        
        System.out.println(max);
        sc.close();
    }
}
