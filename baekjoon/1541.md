import java.util.Scanner;

public class Main {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        String[] subtractionGroups = sc.nextLine().split("-");
        int finalResult = Integer.MAX_VALUE;

        for (int i = 0; i < subtractionGroups.length; i++) {
            int tempSum = 0;

            String[] additionGroups = subtractionGroups[i].split("\\+");
            
            for (String num : additionGroups) {
                tempSum += Integer.parseInt(num);
            }

            if (finalResult == Integer.MAX_VALUE) {
                finalResult = tempSum;
            } else {
                finalResult -= tempSum;
            }
        }
        System.out.println(finalResult);
    }
}
