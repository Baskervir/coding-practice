### 소스 코드
```java
import java.util.*;

public class Main {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        
        if (!sc.hasNextInt()) return;
        int n = sc.nextInt();
        int m = sc.nextInt();
        
        StringBuilder sb = new StringBuilder();
        
        for (int i = 0; i < n; i++) {
            for (int j = 0; j < m; j++) {
                if ((i + j) % 2 == 0) {
                    sb.append('*');
                } else {
                    sb.append('.');
                }
            }
            sb.append('\n');
        }
        
        System.out.print(sb.toString());
        sc.close();
    }
}
```
